/*************************************************
    
    class Node {
        public :

        int data;
        Node *next;

        Node(int data) {
            this -> data = data;
            this -> next = NULL;
        }
    };

*************************************************/

Node* floydAlgorithm(Node* head){
    Node* slow = head;
    Node* fast = head;  // iterate of double speed

    while(slow != NULL && fast != NULL){
        slow = slow->next;
        fast = fast->next->next;
        if(slow == fast){ //means cycle is present
            break;
        }
    }

    // if cycle is present then find starting node of cycle
    if(slow == fast){
        slow = head;
        while(slow->next != fast->next){
            slow = slow->next;
            fast = fast->next;
        }
        // now because slow's next and fast's next are equal after while loop 
        // that means fast is end point of loop or list
        return fast; 
    }

    return NULL;
}

Node *removeLoop(Node *head)
{

    Node* resNode =  floydAlgorithm(head);
    // if no cycle
    if(resNode == NULL){
        return head;
    }

    // if there is cycle
    else {
        if(resNode == head){
            Node* temp = head;
            while(temp->next != head){
                temp = temp->next;
            }
            temp->next = NULL;
        }
        else {
            Node* tail = resNode;
            tail->next = NULL;
        }
        return head;
    }
}
