## My Code:

```cpp
#include <bits/stdc++.h> 
/*************************************************
        Following is the structure of class Node:
     
        class Node{
        public:
            int data;
            Node* next;
            
            Node(int data){
                this->data = data;
                this->next = NULL;
            }
            
        }
**************************************************/

bool isCircular(Node* head){

    map<int, bool> m;
    
    if(head == NULL) return true;

    Node* curr = head;

    while(curr->next != head && curr->next != NULL){

        // this statement i am using specially for test case 3 
        // in which last node pointing to a mid node which is
        // creating a infinite loop 
        // Because each integer value is unique then
        if(m[curr->data]){
            break;
        }
        m[curr->data] = true;
        curr = curr->next;
    }

    if(curr->next == head){
        return true;
    }
    return false;
}
```
