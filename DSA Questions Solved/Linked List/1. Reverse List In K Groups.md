/**
 * Definition for singly-linked list.
 * class Node {
 * public:
 *     int data;
 *     Node *next;
 *     Node() : data(0), next(nullptr) {}
 *     Node(int x) : data(x), next(nullptr) {}
 *     Node(int x, Node *next) : data(x), next(next) {}
 * };
 */

int getLength(Node* &head){
    int crt = 0;
    Node* temp = head;
    
    while(temp != NULL){
        crt++;
        temp = temp->next;
    }
    
    return crt;
}

void solve(Node* &head, Node* prevStartNode, Node* startNode, int remainingLength, int k) {
    
    // base case
    if(startNode == NULL){
        return;
    }
    
    if(remainingLength < k){
        if(startNode != head){
            prevStartNode->next = startNode;
        }
        return;
    }
    

    // processing
    Node* prevNode = NULL;
    Node* currNode = startNode;
    Node* nextNode = NULL;
    
    int pos = 1;
    while(pos<=k){
        nextNode = currNode->next;
        currNode->next = prevNode;
        prevNode = currNode;
        currNode = nextNode;
        pos++;
    }
    
    // intial recurrsion or not 
    if(startNode == head){
        head = prevNode;
    } else {
        prevStartNode->next = prevNode;
    }
    
    solve(head, startNode, nextNode, remainingLength-k, k);

}

Node* kReverse(Node* head, int k) {
    int length = getLength(head);
    solve(head, head, head, length, k);
    return head;
}
